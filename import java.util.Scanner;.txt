import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Hotel hotel = new Hotel();
        Scanner sc = new Scanner(System.in);
        int choice;

        System.out.println("ðŸ¨ Welcome to Cognifyz Hotel Reservation System");

        do {
            System.out.println("\n1. View Rooms\n2. Book Room\n3. Cancel Booking\n4. Exit");
            System.out.print("Choose an option: ");
            choice = sc.nextInt();

            switch (choice) {
                case 1 -> hotel.displayRooms();
                case 2 -> {
                    System.out.print("Enter room number to book: ");
                    hotel.bookRoom(sc.nextInt());
                }
                case 3 -> {
                    System.out.print("Enter room number to cancel: ");
                    hotel.cancelBooking(sc.nextInt());
                }
                case 4 -> System.out.println("Thank you! Goodbye.");
                default -> System.out.println("Invalid option.");
            }
        } while (choice != 4);

        sc.close();
    }
}
